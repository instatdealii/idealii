20  template<int dim>
21  void upwind_temporal_pattern (
22  DoFHandler<dim> &dof ,
23  dealii::DynamicSparsityPattern &time_dsp )
24  {
25  dealii::DoFTools::make_sparsity_pattern ( *dof.temporal () ,
26  time_dsp );
27  if ( dof.fe_support_type () == spacetime::DG_FiniteElement < dim
28  > ::support_type::Lobatto )
29  {
30  for ( dealii::types::global_dof_index ii =
31  dof.dofs_per_cell_time () ;
32  ii < dof.n_dofs_time () ;
33  ii += dof.dofs_per_cell_time () )
34  {
35  time_dsp.add ( ii , ii - 1 );
36  }
37  }
38  else
39  {
40  //go over first DoF of each cell
41  for ( dealii::types::global_dof_index ii =
42  dof.dofs_per_cell_time () ;
43  ii < dof.n_dofs_time () ;
44  ii += dof.dofs_per_cell_time () )
45  {
46  //row offset
47  for ( dealii::types::global_dof_index k = 0 ;
48  k < dof.dofs_per_cell_time () ; k++ )
49  {
50  for ( dealii::types::global_dof_index l = 0 ;
51  l < dof.dofs_per_cell_time () ; l++ )
52  {
53  time_dsp.add ( ii + k , ii - l - 1 );
54  }
55  }
56  }
57  }

173  void make_hanging_node_constraints (
174  idealii::slab::DoFHandler<dim> &dof_handler ,
175  std::shared_ptr<dealii::AffineConstraints<Number>> spacetime_constraints )
176  {
177  auto space_constraints = std::make_shared<dealii::AffineConstraints<Number>> ();
178  dealii::DoFTools::make_hanging_node_constraints (
179  *dof_handler.spatial () ,
180  *space_constraints );
181 
182  unsigned int n_space_dofs = dof_handler.n_dofs_space ();
183  for ( unsigned int time_dof = 0 ; time_dof < dof_handler.n_dofs_time () ; time_dof++ )
184  {
185 
186  for ( unsigned int i = 0 ; i < n_space_dofs ; i++ )
187  {
188  if ( space_constraints->is_constrained ( i ) )
189  {
190  const std::vector<std::pair<dealii::types::global_dof_index,double>> *entries =
191  space_constraints->get_constraint_entries ( i );
192  spacetime_constraints->add_line ( i + time_dof * n_space_dofs );
193  //non Dirichlet constraint
194  if ( entries->size () > 0 )
195  {
196  for ( auto entry : *entries )
197  {
198  spacetime_constraints->add_entry (
199  i + time_dof * n_space_dofs ,
200  entry.first + time_dof * n_space_dofs ,
201  entry.second );
202  }
203  }
204  else
205  {
206  spacetime_constraints->set_inhomogeneity (
207  i + time_dof * n_space_dofs ,
208  space_constraints->get_inhomogeneity ( i ) );
209  }
210  }
211  }
