76  template<int dim,typename Number = double>
77  void interpolate_boundary_values (
78  dealii::IndexSet space_relevant_dofs ,
79  idealii::slab::DoFHandler<dim> &dof_handler ,
80  const dealii::types::boundary_id boundary_component ,
81  dealii::Function<dim,Number> &boundary_function ,
82  std::shared_ptr<dealii::AffineConstraints<Number>> spacetime_constraints ,
83  const dealii::ComponentMask &component_mask = dealii::ComponentMask () )
84  {
85  auto space_constraints = std::make_shared<
86  dealii::AffineConstraints<Number>> ();
87  space_constraints->reinit ( space_relevant_dofs );
88  dealii::Quadrature < 1 > quad_time ( dof_handler.temporal ()->get_fe (0).get_unit_support_points () );
89  dealii::FEValues < 1 > fev (
90  dof_handler.temporal ()->get_fe ( 0 ) ,
91  quad_time ,
92  dealii::update_quadrature_points );
93 
94  unsigned int n_space_dofs = dof_handler.n_dofs_space ();
95  unsigned int time_dof = 0;
96  std::vector < dealii::types::global_dof_index > local_indices (
97  dof_handler.dofs_per_cell_time () );
98  //loop over time cells instead
99  for ( const auto &cell_time : dof_handler.temporal ()->active_cell_iterators () )
100  {
101  fev.reinit ( cell_time );
102 
103  cell_time->get_dof_indices ( local_indices );
104  for ( unsigned int q = 0 ; q < quad_time.size () ; q++ )
105  {
106  space_constraints->clear ();
107  time_dof = local_indices[q];
108  boundary_function.set_time ( fev.quadrature_point ( q )[0] );
109  dealii::VectorTools::interpolate_boundary_values (
110  *dof_handler.spatial () ,
111  boundary_component ,
112  boundary_function ,
113  *space_constraints ,
114  component_mask );
115 
116  for ( auto id = space_relevant_dofs.begin () ; id != space_relevant_dofs.end () ; id++ )
117  {
118  //check if this is a constrained dof
119  if ( space_constraints->is_constrained ( *id ) )
120  {
121  const std::vector<std::pair<dealii::types::global_dof_index, double>> *entries
122  = space_constraints->get_constraint_entries ( *id );
123  spacetime_constraints->add_line ( *id + time_dof * n_space_dofs );
124  //non Dirichlet constraint
125  if ( entries->size () > 0 )
126  {
127  for ( auto entry : *entries )
128  {
129  std::cout << entry.first << ","
130  << entry.second << std::endl;
131  spacetime_constraints->add_entry (
132  *id + time_dof * n_space_dofs ,
133  entry.first + time_dof * n_space_dofs ,
134  entry.second
135  );
136  }
137  }
138  else
139  {
140  spacetime_constraints->set_inhomogeneity (
141  *id + time_dof * n_space_dofs ,
142  space_constraints->get_inhomogeneity ( *id ) );
143  }
144  }
145  }
146  }
147  }
148  }

228  template<int dim>
229  void extract_subvector_at_time_point (
230  slab::DoFHandler<dim> &dof_handler ,
231  const dealii::Vector<double> &spacetime_vector ,
232  dealii::Vector<double> &space_vector ,
233  const double t )
234  {
235  space_vector = 0;
236  unsigned int n_dofs_space = space_vector.size ();
237  double left = 0;
238  double right = 0;
239  for ( auto cell : dof_handler.temporal ()->active_cell_iterators () )
240  {
241  left = cell->face ( 0 )->center () ( 0 );
242  right = cell->face ( 1 )->center () ( 0 );
243  if ( t < left || t > right )
244  {
245  continue;
246  }
247  double _t = ( t - left ) / ( right - left );
248  dealii::Point < 1 > qpoint ( _t );
249  const dealii::Quadrature<1> time_qf ( qpoint );
250  dealii::FEValues < 1 > time_values (
251  dof_handler.temporal ()->get_fe () ,
252  time_qf ,
253  dealii::update_values );
254  time_values.reinit ( cell );
255  unsigned int offset = cell->index ()
256  * dof_handler.dofs_per_cell_time ()
257  * n_dofs_space;
258  for ( unsigned int ii = 0 ; ii < dof_handler.dofs_per_cell_time () ; ii++ )
259  {
260  double factor = time_values.shape_value ( ii , 0 );
261  for ( unsigned int i = 0 ; i < n_dofs_space ; i++ )
262  {
263  space_vector[i] += spacetime_vector[i + ii * n_dofs_space + offset] * factor;
264  }
265  }
266  }
267  }
